// ==UserScript==
// @name         RhythmVerse Clone Hero Ownership Checker
// @namespace    http://tampermonkey.net/
// @version      2.4
// @description  Batch scanning, deterministic IDs, dynamically change folder via UI button
// @author       Wavez
// @match        *://*.rhythmverse.co/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    console.log("🟢 RV SCRIPT: Injected and running with Floating Settings Button!");

    const API_URL = "http://127.0.0.1:5000/check";
    const BATCH_URL = "http://127.0.0.1:5000/check_batch";
    const CARD_SELECTOR = 'tr';
    const TITLE_SELECTOR = 'a[href*="/songfile"]';
    const ARTIST_SELECTOR = 'a[href*="/songfile"] + span';

    const checkedCards = new Set();
    let isServerDown = false;
    let isProcessing = false;
    let isInitialLoad = true;

    // --- NEW: Inject a Floating "Change Folder" Button ---
    const settingsBtn = document.createElement('button');
    settingsBtn.innerHTML = '⚙️ Change Local Folder';
    settingsBtn.style.position = 'fixed';
    settingsBtn.style.bottom = '20px';
    settingsBtn.style.right = '20px';
    settingsBtn.style.zIndex = '9999';
    settingsBtn.style.padding = '12px 20px';
    settingsBtn.style.backgroundColor = '#1f2937';
    settingsBtn.style.color = '#4ade80';
    settingsBtn.style.border = '2px solid #374151';
    settingsBtn.style.borderRadius = '8px';
    settingsBtn.style.cursor = 'pointer';
    settingsBtn.style.fontWeight = 'bold';
    settingsBtn.style.boxShadow = '0 4px 6px rgba(0,0,0,0.5)';
    settingsBtn.style.transition = 'all 0.2s';

    settingsBtn.onmouseover = () => {
        settingsBtn.style.backgroundColor = '#374151';
        settingsBtn.style.transform = 'scale(1.05)';
    };
    settingsBtn.onmouseout = () => {
        settingsBtn.style.backgroundColor = '#1f2937';
        settingsBtn.style.transform = 'scale(1)';
    };

    settingsBtn.onclick = async () => {
        settingsBtn.innerHTML = '⏳ Check your Taskbar for Popup...';
        try {
            const response = await fetch('http://127.0.0.1:5000/change_folder');
            const data = await response.json();

            if (data.status === 'success') {
                alert(`Folder updated successfully to:\n${data.folder}\n\nThe page will now refresh to scan your new library!`);
                location.reload();
            } else {
                settingsBtn.innerHTML = '⚙️ Change Local Folder';
            }
        } catch (e) {
            alert("Ensure your Python scanner app is running first!");
            settingsBtn.innerHTML = '⚙️ Change Local Folder';
        }
    };
    document.body.appendChild(settingsBtn);
    // -----------------------------------------------------

    function getCleanText(el) {
        let text = "";
        for (let node of el.childNodes) {
            if (node.nodeType === Node.TEXT_NODE) {
                text += node.textContent;
            }
        }
        return text.trim();
    }

    function setOwnedUI(card, titleEl) {
        card.style.backgroundColor = "rgba(74, 222, 128, 0.15)";
        let badge = titleEl.querySelector('.rv-status-badge');
        if (!badge) {
            badge = document.createElement('span');
            badge.className = 'rv-status-badge';
            badge.style.marginLeft = "8px";
            titleEl.appendChild(badge);
        }
        badge.innerText = " ✅ OWNED";
        badge.style.color = "#16a34a";
        badge.style.fontSize = "0.85em";
        badge.style.fontWeight = "bold";
    }

    function trackDownload(card, titleEl, artist, title, rowId) {
        if (card.dataset.isTracking === "true") return;
        card.dataset.isTracking = "true";
        card.style.backgroundColor = "rgba(234, 179, 8, 0.15)";

        let badge = titleEl.querySelector('.rv-status-badge');
        if (!badge) {
            badge = document.createElement('span');
            badge.className = 'rv-status-badge';
            badge.style.marginLeft = "8px";
            titleEl.appendChild(badge);
        }
        badge.innerText = " ⏳ DOWNLOADING / EXTRACTING...";
        badge.style.color = "#ca8a04";
        badge.style.fontSize = "0.85em";
        badge.style.fontWeight = "bold";

        fetch(`${API_URL}?id=${encodeURIComponent(rowId)}&artist=${encodeURIComponent(artist)}&title=${encodeURIComponent(title)}&downloading=true`)
            .catch(err => {});

        const pollInterval = setInterval(async () => {
            try {
                const response = await fetch(`${API_URL}?id=${encodeURIComponent(rowId)}&artist=${encodeURIComponent(artist)}&title=${encodeURIComponent(title)}&polling=true`);
                const data = await response.json();

                if (data.owned) {
                    clearInterval(pollInterval);
                    document.querySelectorAll(CARD_SELECTOR).forEach(matchingCard => {
                        const matchTitleEl = matchingCard.querySelector(TITLE_SELECTOR);
                        const matchArtistEl = matchingCard.querySelector(ARTIST_SELECTOR);
                        if (matchTitleEl && matchArtistEl) {
                            if (getCleanText(matchTitleEl) === title && getCleanText(matchArtistEl) === artist) {
                                setOwnedUI(matchingCard, matchTitleEl);
                                matchingCard.dataset.isTracking = "true";
                            }
                        }
                    });
                }
            } catch (err) {}
        }, 3000);
    }

    async function processBatch() {
        if (isProcessing) return;
        isProcessing = true;

        const allCardsInDOM = Array.from(document.querySelectorAll(CARD_SELECTOR));

        let rows = allCardsInDOM.filter(card => {
            return !checkedCards.has(card) && card.getBoundingClientRect().height > 0 && card.offsetParent;
        });

        if (rows.length === 0) {
            isProcessing = false;
            return;
        }

        rows.sort((a, b) => a.getBoundingClientRect().top - b.getBoundingClientRect().top);

        const payload = [];
        const validRows = [];

        for (let card of rows) {
            const titleEl = card.querySelector(TITLE_SELECTOR);
            const artistEl = card.querySelector(ARTIST_SELECTOR);

            if (titleEl && artistEl) {
                checkedCards.add(card);

                const cleanArtist = getCleanText(artistEl);
                const cleanTitle = getCleanText(titleEl);

                let occurrenceCount = 0;
                const cardIndex = allCardsInDOM.indexOf(card);
                for (let i = 0; i < cardIndex; i++) {
                    const prevTitleEl = allCardsInDOM[i].querySelector(TITLE_SELECTOR);
                    const prevArtistEl = allCardsInDOM[i].querySelector(ARTIST_SELECTOR);
                    if (prevTitleEl && prevArtistEl) {
                        if (getCleanText(prevTitleEl) === cleanTitle && getCleanText(prevArtistEl) === cleanArtist) {
                            occurrenceCount++;
                        }
                    }
                }

                const rowId = `${cleanArtist}|${cleanTitle}|${occurrenceCount}`;

                validRows.push({ card, titleEl, artist: cleanArtist, title: cleanTitle, id: rowId });
                payload.push({ id: rowId, artist: cleanArtist, title: cleanTitle });
            }
        }

        if (payload.length === 0) {
            isProcessing = false;
            return;
        }

        try {
            const response = await fetch(BATCH_URL, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    items: payload,
                    is_initial_load: isInitialLoad
                })
            });
            const data = await response.json();

            isInitialLoad = false;

            if (isServerDown) {
                console.log("🟢 RV SCRIPT: Connection to Python Server restored!");
                isServerDown = false;
            }

            for (let i = 0; i < validRows.length; i++) {
                const rowData = validRows[i];
                const result = data[i];

                if (result.owned) {
                    setOwnedUI(rowData.card, rowData.titleEl);
                } else {
                    const downloadLinks = rowData.card.querySelectorAll('a[href*="/download"]');
                    downloadLinks.forEach(btn => {
                        const handleDownloadTrigger = (e) => {
                            if (e.type === 'click' || (e.type === 'auxclick' && e.button === 1)) {
                                trackDownload(rowData.card, rowData.titleEl, rowData.artist, rowData.title, rowData.id);
                            }
                        };
                        btn.addEventListener('click', handleDownloadTrigger);
                        btn.addEventListener('auxclick', handleDownloadTrigger);
                    });
                }
            }
        } catch (err) {
            for (let row of validRows) {
                checkedCards.delete(row.card);
            }
            if (!isServerDown) {
                console.error("🔴 RV SCRIPT: Python server offline. Waiting for it to start...");
                isServerDown = true;
            }
        }

        isProcessing = false;
    }

    setInterval(processBatch, 2000);
    setTimeout(processBatch, 500);

})();
